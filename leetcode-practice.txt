var romanToInt = function(s){
  const romanNums = {
    I: 1,
    V: 5,
    X: 10,
    L: 50,
    C: 100,
    D: 500,
    M: 1000
  }
  let result = 0;
  for(let i = 0; i < s.length; i++){
    romanNums[s[i]] < romanNums[s[i + 1]] ? result -= romanNums[s[i]]:
    result += romanNums[s[i]]
  }
    return result;
}
